3. CODE LOGIC:
```javascript
   const processUploadedImages = async (images) => {
     const FREE_AI_LIMIT = 5;
     const items = [];
     
     for (let i = 0; i < images.length; i++) {
       if (i < FREE_AI_LIMIT) {
         // First 5: Use AI
         const aiData = await generateAIDescription(images[i]);
         items.push({
           imageUrl: images[i],
           title: aiData.title,
           description: aiData.description,
           category: aiData.category,
           price: aiData.usedPrice,
           condition: aiData.condition,
           isAIGenerated: true
         });
       } else {
         // Items 6+: Empty form
         items.push({
           imageUrl: images[i],
           title: '',
           description: '',
           category: '',
           price: '',
           condition: '',
           isAIGenerated: false
         });
       }
     }
     
     return items;
   };
