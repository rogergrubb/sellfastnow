javascriptconst handleAIUpgrade = async () => {
  const unpopulatedItems = items.filter(item => !item.hasAI);
  const cost = unpopulatedItems.length * 0.20;
  
  // Show payment modal
  const paymentConfirmed = await showStripePayment(cost);
  
  if (paymentConfirmed) {
    // Show progress modal
    setShowProgress(true);
    
    // Generate AI for remaining items
    for (let i = 0; i < unpopulatedItems.length; i++) {
      updateProgress(`Generating item ${i + 1} of ${unpopulatedItems.length}...`);
      const aiData = await generateAIDescription(unpopulatedItems[i].imageUrl);
      updateItemWithAI(unpopulatedItems[i].id, aiData);
    }
    
    setShowProgress(false);
    showSuccessToast('All items completed with AI!');
  }
};
