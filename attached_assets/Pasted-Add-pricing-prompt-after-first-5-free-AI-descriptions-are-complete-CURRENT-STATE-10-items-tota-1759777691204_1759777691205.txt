  // Show pricing modal if there are remaining items
  if (groupedImages.length > FREE_LIMIT) {
    const remaining = groupedImages.length - FREE_LIMIT;
    showPricingModal(remaining);
  } else {
    // All items covered by free tier, go straight to review
    showReviewForm(processedItems);
  }
};
